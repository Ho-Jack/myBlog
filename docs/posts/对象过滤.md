## 过滤对象及数组的方法

### 向对象添加元素

 #### 第一种方法

 ```js
 let obj ={"name":"tom","age":16}
 let key = "id";
 let value = 2
 obj[key] = value;
 //obj  {name: "tom", age: 16, id: 2}
 ```

 #### 第二种方法，利用扩展运算符,简单又实用

 ```js
var obj1={"vue":300,"jquery":200};
 var obj2={"react":500};
 var obj3={
 	...obj1,
 	...obj2
 }
 //obj3  {vue: 300, jquery: 200, react: 500}
 ```

 #### 第三种办法

 ```js
let obj4 = {a:"1"};
 let obj5 = {b:"2"};
let obj6 = {c:"3"};
 Object.assign(obj4,obj5,obj6);
 //obj4 {a: "1", b: "2", c: "3"}
 ```



#### 过滤对象,整合成新对象

```js
  var obj= {
    偃师市: '80%',
    伊川县: '80%',
    洛宁县: '80%',
    宜阳县: '80%',
    汝阳县: '80%',
  }

  var y = {}
  for (const key in obj) {
   y[key] = obj[key].split('%')[0]
      //键：值
  }

  console.log(y)
    最终结果：
   y={
    偃师市: '80',
    伊川县: '80',
    洛宁县: '80',
    宜阳县: '80',
    汝阳县: '80',
   }
```

#### 过滤对象，整合成新的数组

```js
	var index = 0
			var y = []
			var mapTooltip = {
                "偃师市": "1",
                "伊川县": "9",
                "洛宁县": "34",
              }
			var tooltipColor = ['#72CFDA', '#0648B3', '#ADDE62']
			for (const key in mapTooltip) {
				y[index] = { 
                             name: key,
                             value: mapTooltip[key],
                             color: tooltipColor[index]
                           }
                // 索引：值   
				index++
			}	
```
